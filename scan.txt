#include<stdio.h>
int main()
{
int q[10],q1[10],q2[10],mrange,head,size,a,b,c,i,j,seek=0,diff;
float avg;
printf("enter max range of disk\n");
scanf("%d",&mrange);
printf("enter head position of disk\n");
scanf("%d",&head);
printf("enter size of queue\n");
scanf("%d",&size);
printf("enter disk positions to be read\n");
b=0;
c=0;
for(i=0;i<size;i++)
{
scanf("%d",&a);
if(a>head)
{
q1[b]=a;
b++;
}
else
{
q2[c]=a;
c++;
}
}
for(i=0;i<b-1;i++)
{
for(j=i+1;j<b;j++)
{
if(q1[i]>q1[j])
{
a=q1[i];
q1[i]=q1[j];
q1[j]=a;
}
}
}
for(i=0;i<c-1;i++)
{
for(j=1;j<c;j++)
{
if(q2[i]<q2[j])
{
a=q2[i];
q2[i]=q2[j];
q2[j]=a;
}
}
}

for(i=1,j=0;j<b;i++,j++)
q[i]=q1[j];
q[i]=mrange;
for(i=b+2,j=0;j<c;i++,j++)
q[i]=q2[j];
q[i]=0;
q[0]=head;
for(i=0;i<=size+1;i++)
{
diff=abs(q[i+1]-q[i]);
seek=seek+diff;
printf("disk head moves from %d to %d with seek %d \n",q[i],q[i+1],diff);
}
avg=seek/size;
printf("total seek time is %d\n",seek);
printf("avg seek time is %f\n",avg);
}
